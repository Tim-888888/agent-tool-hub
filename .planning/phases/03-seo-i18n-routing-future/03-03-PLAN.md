    **1. Create `src/app/sitemap.ts`:**

    ```typescript
    import type { MetadataRoute } from 'next';
    import { prisma } from '@/lib/db';

    const BASE_URL = process.env.NEXT_PUBLIC_APP_URL || 'https://agenttoolhub.com';

    export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
      // Fetch all active tools and categories from database
      const tools = await prisma.tool.findMany({
        where: { status: { in: ['ACTIVE', 'FEATURED'] } },
        select: { slug: true, updatedAt: true },
      });

      const categories = await prisma.category.findMany({
        select: { slug: true, updatedAt: true },
      });

      // Tool URLs with locale alternates
      const toolUrls: MetadataRoute.Sitemap = tools.map((tool) => ({
        url: `${BASE_URL}/tools/${tool.slug}`,
        lastModified: tool.updatedAt,
        changeFrequency: 'weekly' as const,
        priority: 0.8,
        alternates: {
          languages: {
            zh: `${BASE_URL}/zh/tools/${tool.slug}`,
          },
        },
      }));

      // Category URLs with locale alternates
      const categoryUrls: MetadataRoute.Sitemap = categories.map((cat) => ({
        url: `${BASE_URL}/categories/${cat.slug}`,
        lastModified: cat.updatedAt,
        changeFrequency: 'weekly' as const,
        priority: 0.6,
        alternates: {
          languages: {
            zh: `${BASE_URL}/zh/categories/${cat.slug}`,
          },
        },
      }));

      // Static pages with locale alternates
      const staticPages: MetadataRoute.Sitemap = [
        {
          url: BASE_URL,
          lastModified: new Date(),
          changeFrequency: 'daily' as const,
          priority: 1.0,
          alternates: {
            languages: {
              zh: `${BASE_URL}/zh`,
            },
          },
        },
        {
          url: `${BASE_URL}/tools`,
          lastModified: new Date(),
          changeFrequency: 'daily' as const,
          priority: 0.9,
          alternates: {
            languages: {
              zh: `${BASE_URL}/zh/tools`,
            },
          },
        },
        {
          url: `${BASE_URL}/rankings`,
          lastModified: new Date(),
          changeFrequency: 'daily' as const,
          priority: 0.7,
          alternates: {
            languages: {
              zh: `${BASE_URL}/zh/rankings`,
            },
          },
        },
        {
          url: `${BASE_URL}/submit`,
          lastModified: new Date(),
          changeFrequency: 'monthly' as const,
          priority: 0.5,
          alternates: {
            languages: {
              zh: `${BASE_URL}/zh/submit`,
            },
          },
        },
        {
          url: `${BASE_URL}/about`,
          lastModified: new Date(),
          changeFrequency: 'monthly' as const,
          priority: 0.4,
          alternates: {
            languages: {
              zh: `${BASE_URL}/zh/about`,
            },
          },
        },
      ];

      return [...staticPages, ...toolUrls, ...categoryUrls];
    }
    ```

    Key decisions:
    - Tool pages get priority 0.8 (high value content)
    - Category pages get priority 0.6
    - Home page gets priority 1.0
    - All pages include `alternates.languages` with `zh` variant for hreflang
    - Only ACTIVE/FEATURED tools included (excludes PENDING/ARCHIVED)
    - changeFrequency: 'weekly' for content pages, 'daily' for lists, 'monthly' for static

    **2. Create `src/app/robots.ts`:**

    ```typescript
    import type { MetadataRoute } from 'next';

    const BASE_URL = process.env.NEXT_PUBLIC_APP_URL || 'https://agenttoolhub.com';

    export default function robots(): MetadataRoute.Robots {
      return {
        rules: {
          userAgent: '*',
          allow: '/',
          disallow: '/api/',
        },
        sitemap: `${BASE_URL}/sitemap.xml`,
      };
    }
    ```

    This produces:
    ```
    User-agent: *
    Allow: /
    Disallow: /api/

    Sitemap: https://agenttoolhub.com/sitemap.xml
    ```

    **Important note on proxy.ts interaction:**
    The proxy matcher already excludes sitemap.xml and robots.txt:
    `['/((?!api|_next/static|_next/image|favicon.ico|sitemap.xml|robots.txt).*)']`
    So these routes will not be intercepted by the locale detection logic.
