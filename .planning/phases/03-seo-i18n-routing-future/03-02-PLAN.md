    function buildJsonLd(tool: Tool, locale: string): object {
      const isZh = locale === 'zh';
      const BASE_URL = process.env.NEXT_PUBLIC_APP_URL || 'https://agenttoolhub.com';
      const localePath = isZh ? '/zh' : '';

      return {
        '@context': 'https://schema.org',
        '@type': 'SoftwareApplication',
        name: tool.name,
        description: isZh && tool.descriptionZh ? tool.descriptionZh : tool.description,
        url: `${BASE_URL}${localePath}/tools/${tool.slug}`,
        applicationCategory: 'DeveloperApplication',
        operatingSystem: 'Cross-platform',
        // Per TECHNICAL_DESIGN.md 2.3: installInstructions array
        installInstructions: buildInstallInstructions(tool),
        // Requirements extracted from tool metadata
        requirements: buildRequirements(tool),
        homepage: tool.repoUrl,
        offers: {
          '@type': 'Offer',
          price: '0',
          priceCurrency: 'USD',
        },
        aggregateRating: {
          '@type': 'AggregateRating',
          ratingValue: tool.avgRating.toFixed(1),
          reviewCount: tool.ratingCount,
          bestRating: '5',
          worstRating: '1',
        },
        programmingLanguage: tool.language ?? undefined,
        license: tool.license ?? undefined,
        author: tool.author

      };
    }

    **Create helper functions:**

    ```typescript
    function buildInstallInstructions(tool: Tool): Array<{ platform: string; command: string }> {
      if (!tool.installGuide) return [];
      try {
        const guide = typeof tool.installGuide === 'string'
          ? JSON.parse(tool.installGuide)
          : tool.installGuide;
        // installGuide contains per-platform install data
        // Return structured array for schema.org
        const instructions: Array<{ platform: string; command: string }> = [];
        if (guide.claudeCode) instructions.push({ platform: 'Claude Code', command: guide.claudeCode });
        if (guide.cursor) instructions.push({ platform: 'Cursor', command: guide.cursor });
        if (guide.cline) instructions.push({ platform: 'Cline', command: guide.cline });
        if (guide.windsurf) instructions.push({ platform: 'Windsurf', command: guide.windsurf });
        if (guide.vscode) instructions.push({ platform: 'VS Code', command: guide.vscode });
        return instructions;
      } catch {
        return [];
      }
    }

    function buildRequirements(tool: Tool): string[] {
      const requirements: string[] = [];
      if (tool.language) requirements.push(`Runtime: ${tool.language}`);
      if (tool.license) requirements.push(`License: ${tool.license}`);
      requirements.push('Node.js >= 18');
      return requirements;
    }
    ```
