import { PrismaClient } from '@prisma/client';

const globalForPrisma = globalThis as unknown as { prisma: PrismaClient };

export const prisma =
  globalForPrisma.prisma ||
  new PrismaClient({
    log: process.env.NODE_ENV === 'development' ? ['query', 'error', 'warn'] : ['error'],

  <action>
Create `prisma/seed.ts` that mirrors ALL data from `src/lib/mock-data.ts` into the database using Prisma `upsert` for idempotency.

Structure:
1. Import `PrismaClient` and enums from `@prisma/client`
2. Define PLATFORMS, CATEGORIES, and TOOLS arrays inline (copy data from mock-data.ts) — do NOT import from `src/lib/mock-data.ts` because the seed runs outside Next.js and the mock-data module uses `@/` path aliases that may not resolve in tsx
3. Seed platforms: loop over PLATFORMS array, use `prisma.platform.upsert({ where: { slug }, update: { name, icon, configKey }, create: fullObject })`
4. Seed categories: loop over CATEGORIES array, use `prisma.category.upsert({ where: { slug }, update: { nameEn, nameZh, icon, descriptionEn, descriptionZh, order }, create: fullObject })`
5. Seed tools with relations:
   - For each tool: destructure `{ categories: toolCats, platforms: toolPlats, ...toolFields }` from the tool data
   - Use `prisma.tool.upsert({ where: { slug }, update: toolFields, create: toolFields })` — the toolFields object must match the Prisma model shape (exclude categories and platforms arrays from direct fields)
   - For date fields (lastCommitAt, createdAt, updatedAt): pass as `new Date("...")` objects, not raw strings
   - For installGuide: pass the JSON object directly (Prisma Json type accepts plain objects)
   - After tool upsert, connect categories: for each category slug, find the category record, then `prisma.toolCategory.upsert({ where: { toolId_categoryId: { toolId, categoryId } }, update: {}, create: { toolId, categoryId } })`
   - Connect platforms the same way using `prisma.toolPlatform.upsert`
6. Wrap in `main()` function with try/catch, `prisma.$disconnect()` in finally block
7. Log progress: "Seeding platforms...", "Seeding categories...", "Seeding tools...", "Seeding complete. Created X tools, Y categories, Z platforms."

CRITICAL details:
- Use `ToolType.MCP_SERVER` etc. from Prisma enums (imported from `@prisma/client`), not string literals
- Tool status `FEATURED` and `ACTIVE` must use `ToolStatus.FEATURED` / `ToolStatus.ACTIVE`
- The `installGuide` field is `Json?` — pass the object directly, NOT JSON.stringify()
- The `tags`, `transports`, `featuresZh`, `featuresEn`, `screenshots` fields are `String[]` — pass arrays directly
- Skip `id` field in seed — Prisma auto-generates cuid() ids
